# Run All Tests
# Master test runner that executes all tests and shows report

# Initialize test counters
SetVariable: $$testResults = ""
SetVariable: $$testsPassed = 0
SetVariable: $$testsFailed = 0
SetVariable: $$testsTotal = 0

# Show starting dialog
ShowDialog: "PDF/A Test Suite" "Running comprehensive tests...¶This may take a moment."

# Run all individual tests
PerformScript: "Test 1: Success Case - Valid PDF"
PerformScript: "Test 2: Error - Empty Container"
PerformScript: "Test 3: Error - Invalid Output Path"
PerformScript: "Test 4: Error - File Not Found"
PerformScript: "Test 5: Success - With File Path"
PerformScript: "Test 6: Success - With Metadata"

# Build report
SetVariable: $report = "=== PDF/A Test Suite Results ===" & ¶ & "Total Tests: " & $$testsTotal & ¶ & "Passed: ✓ " & $$testsPassed & ¶ & "Failed: ✗ " & $$testsFailed & ¶ & "Success Rate: " & Round($$testsPassed / $$testsTotal * 100; 0) & "%" & ¶ & ¶ & "=== Detailed Results ===" & ¶ & $$testResults

# Show final report
ShowDialog: "Test Suite Complete" $report
